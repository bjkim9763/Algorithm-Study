```cpp
#include <iostream>

using namespace std;

int N;
int ans = 0;

int main(void) {
	cin >> N;

	// ------------------------------------
	// 
	// 5kg의 봉지로만 설탕을 배달할 때 가장 적은 수의 봉지를 든다.
	// 5kg의 봉지만으로는 설탕을 배달할 수 없을 때 3kg 봉지를 하나씩 추가하는 것이 가장 적은 수의 봉지를 드는 방법이다.
	// 
	// ------------------------------------

	while (N>=0) {
		if (N % 5 == 0) // 5kg의 봉지만으로 설탕을 배달할 수 있는가?
		{
			ans += (N / 5);
			cout << ans;
			return 0;
		}
		else			// 불가능하다면 3kg 봉지를 추가해본다
		{
			ans++;		// 봉지를 늘린다.
			N -= 3;		// N을 3 빼준다. 빼준 N을 가지고 반복을 돌면서 한번 더 5kg의 계산을 해본다.
		}
	}
	
	cout << -1;

	return 0;
}
```
